```python
import torch
import torch.nn as nn
import math

class MultiHeadAttention(nn.Module):
    """
    多头自注意力机制
    考点：理解 Q/K/V 计算、缩放点积、多头拼接
    """
    def __init__(self, d_model: int, num_heads: int, dropout: float = 0.1):
        super().__init__()
        assert d_model % num_heads == 0, "d_model must be divisible by num_heads"
        
        self.d_model = d_model
        self.num_heads = num_heads
        self.d_k = d_model // num_heads
        
        # 线性投影层
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.W_o = nn.Linear(d_model, d_model)
        
        self.dropout = nn.Dropout(dropout)
        self.scale = math.sqrt(self.d_k)
    
    def forward(self, q: torch.Tensor, k: torch.Tensor, v: torch.Tensor, 
                mask: torch.Tensor = None) -> torch.Tensor:
        """
        Args:
            q: [batch_size, seq_len_q, d_model]
            k: [batch_size, seq_len_k, d_model]
            v: [batch_size, seq_len_v, d_model]
            mask: [batch_size, 1, seq_len_q, seq_len_k] 或 None
        Returns:
            output: [batch_size, seq_len_q, d_model]
        """
        batch_size = q.size(0)
        
        # 1. 线性投影并分割多头
        # [batch, seq_len, d_model] -> [batch, seq_len, num_heads, d_k]
        Q = self.W_q(q).view(batch_size, -1, self.num_heads, self.d_k)
        K = self.W_k(k).view(batch_size, -1, self.num_heads, self.d_k)
        V = self.W_v(v).view(batch_size, -1, self.num_heads, self.d_k)
        
        # [batch, seq_len, num_heads, d_k] -> [batch, num_heads, seq_len, d_k]
        Q = Q.transpose(1, 2)
        K = K.transpose(1, 2)
        V = V.transpose(1, 2)
        
        # 2. 计算注意力分数：Q @ K^T / sqrt(d_k)
        # [batch, num_heads, seq_q, d_k] @ [batch, num_heads, d_k, seq_k]
        # -> [batch, num_heads, seq_q, seq_k]
        scores = torch.matmul(Q, K.transpose(-2, -1)) / self.scale
        
        # 3. 应用 mask（如果有）
        if mask is not None:
            scores = scores.masked_fill(mask == 0, float('-inf'))
        
        # 4. Softmax + Dropout
        attn_weights = torch.softmax(scores, dim=-1)  # [batch, num_heads, seq_q, seq_k]
        attn_weights = self.dropout(attn_weights)
        
        # 5. 加权求和：attn @ V
        # [batch, num_heads, seq_q, seq_k] @ [batch, num_heads, seq_k, d_k]
        # -> [batch, num_heads, seq_q, d_k]
        context = torch.matmul(attn_weights, V)
        
        # 6. 拼接多头并输出投影
        # [batch, num_heads, seq_q, d_k] -> [batch, seq_q, num_heads, d_k]
        context = context.transpose(1, 2).contiguous()
        context = context.view(batch_size, -1, self.d_model)
        
        output = self.W_o(context)  # [batch, seq_q, d_model]
        
        return output, attn_weights
```

#### 题目 2：实现 Cross-Attention（多模态融合关键）

```python
class CrossAttention(nn.Module):
    """
    跨模态注意力机制
    用于多模态融合：Q 来自文本，K/V 来自图像/视频
    考点：理解 Q/K/V 来源不同模态的处理
    """
    def __init__(self, d_model: int, num_heads: int, dropout: float = 0.1):
        super().__init__()
        self.d_model = d_model
        self.num_heads = num_heads
        self.d_k = d_model // num_heads
        
        # Q 来自 target 模态（如文本）
        self.W_q = nn.Linear(d_model, d_model)
        # K, V 来自 source 模态（如图像）
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.W_o = nn.Linear(d_model, d_model)
        
        self.dropout = nn.Dropout(dropout)
        self.scale = math.sqrt(self.d_k)
        self.layer_norm = nn.LayerNorm(d_model)
    
    def forward(self, query: torch.Tensor, key_value: torch.Tensor, 
                mask: torch.Tensor = None) -> torch.Tensor:
        """
        Args:
            query: [batch_size, seq_len_q, d_model] - 文本 token
            key_value: [batch_size, seq_len_kv, d_model] - 图像 patch
            mask: 可选的注意力 mask
        Returns:
            output: [batch_size, seq_len_q, d_model]
        """
        batch_size = query.size(0)
        residual = query  # 残差连接
        
        # 归一化
        query = self.layer_norm(query)
        
        # 线性投影
        Q = self.W_q(query).view(batch_size, -1, self.num_heads, self.d_k).transpose(1, 2)
        K = self.W_k(key_value).view(batch_size, -1, self.num_heads, self.d_k).transpose(1, 2)
        V = self.W_v(key_value).view(batch_size, -1, self.num_heads, self.d_k).transpose(1, 2)
        
        # 注意力计算
        scores = torch.matmul(Q, K.transpose(-2, -1)) / self.scale
        if mask is not None:
            scores = scores.masked_fill(mask == 0, float('-inf'))
        
        attn_weights = torch.softmax(scores, dim=-1)
        attn_weights = self.dropout(attn_weights)
        
        # 加权求和
        context = torch.matmul(attn_weights, V)
        context = context.transpose(1, 2).contiguous().view(batch_size, -1, self.d_model)
        
        # 输出投影 + 残差
        output = self.W_o(context) + residual
        
        return output
```

#### 题目 3：实现位置编码（Positional Encoding）

```python
class PositionalEncoding(nn.Module):
    """
    正弦位置编码（原始 Transformer）
    考点：理解为什么需要位置编码、正弦编码的优势
    """
    def __init__(self, d_model: int, max_len: int = 5000, dropout: float = 0.1):
        super().__init__()
        self.dropout = nn.Dropout(dropout)
        
        # 创建位置编码矩阵 [max_len, d_model]
        pe = torch.zeros(max_len, d_model)
        
        # 位置 [max_len, 1]
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        
        # 计算除数：10000^(2i/d_model)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * 
                            (-math.log(10000.0) / d_model))
        
        # 正弦编码偶数维度，余弦编码奇数维度
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        
        # 注册为 buffer（不作为参数更新）
        self.register_buffer('pe', pe.unsqueeze(0))  # [1, max_len, d_model]
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: [batch_size, seq_len, d_model]
        Returns:
            x + positional_encoding
        """
        # x: [batch, seq_len, d_model]
        # pe: [1, max_len, d_model]
        # 取前 seq_len 个位置
        pe = self.pe[:, :x.size(1), :]
        return self.dropout(x + pe)


class LearnablePositionalEncoding(nn.Module):
    """
    可学习的位置编码（BERT、ViT 使用）
    考点：对比正弦编码与可学习编码的优劣
    """
    def __init__(self, d_model: int, max_len: int = 512, dropout: float = 0.1):
        super().__init__()
        self.dropout = nn.Dropout(dropout)
        # 可学习的位置 embedding
        self.pe = nn.Parameter(torch.randn(1, max_len, d_model) * 0.02)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        pe = self.pe[:, :x.size(1), :]
        return self.dropout(x + pe)


class RotaryPositionalEmbedding(nn.Module):
    """
    RoPE 旋转位置编码（LLaMA、Qwen 使用）
    考点：理解相对位置编码、旋转矩阵
    """
    def __init__(self, d_model: int, max_len: int = 2048):
        super().__init__()
        assert d_model % 2 == 0
        
        # 计算旋转角度
        inv_freq = 1.0 / (10000 ** (torch.arange(0, d_model, 2).float() / d_model))
        self.register_buffer('inv_freq', inv_freq)
        
        self.max_len = max_len
        self.d_model = d_model
    
    def forward(self, x: torch.Tensor, seq_dim: int = 1) -> torch.Tensor:
        """
        Args:
            x: [batch, seq_len, num_heads, d_k] 或 [batch, seq_len, d_model]
        Returns:
            x 应用 RoPE 后的结果
        """
        seq_len = x.size(seq_dim)
        
        # 计算位置 [seq_len]
        t = torch.arange(seq_len, device=x.device).type_as(self.inv_freq)
        
        # 计算角度 [seq_len, d_model/2]
        freqs = torch.einsum('i,j->ij', t, self.inv_freq)
        
        # 计算旋转的复数形式
        emb = torch.cat((freqs, freqs), dim=-1)
        cos = emb.cos()
        sin = emb.sin()
        
        # 应用旋转到 x
        # 这里简化实现，实际需要考虑多头情况
        if x.dim() == 4:  # [batch, seq, heads, d_k]
            x1, x2 = x[..., ::2], x[..., 1::2]
            x_rotated = torch.stack((-x2, x1), dim=-1).flatten(-2)
        else:  # [batch, seq, d_model]
            x1, x2 = x[..., ::2], x[..., 1::2]
            x_rotated = torch.cat((-x2, x1), dim=-1)
        
        return x * cos + x_rotated * sin
```

```python
class SimplifiedCLIP(nn.Module):
    """
    简化版 CLIP 模型
    考点：理解对比学习、双塔架构、温度参数
    """
    def __init__(self, image_encoder: nn.Module, text_encoder: nn.Module, 
                 embed_dim: int = 512, temperature: float = 0.07):
        super().__init__()
        self.image_encoder = image_encoder
        self.text_encoder = text_encoder
        
        # 投影层
        self.image_projection = nn.Parameter(torch.empty(embed_dim, embed_dim))
        self.text_projection = nn.Parameter(torch.empty(embed_dim, embed_dim))
        
        # 可学习的温度参数
        self.logit_scale = nn.Parameter(torch.log(torch.tensor(1 / temperature)))
        
        self._init_weights()
    
    def _init_weights(self):
        nn.init.normal_(self.image_projection, std=0.02)
        nn.init.normal_(self.text_projection, std=0.02)
    
    def encode_image(self, images: torch.Tensor) -> torch.Tensor:
        """
        Args:
            images: [batch_size, 3, H, W]
        Returns:
            image_features: [batch_size, embed_dim] 归一化后
        """
        # 通过图像编码器
        image_features = self.image_encoder(images)
        # 投影
        image_features = image_features @ self.image_projection
        # L2 归一化
        image_features = nn.functional.normalize(image_features, p=2, dim=-1)
        return image_features
    
    def encode_text(self, texts: torch.Tensor) -> torch.Tensor:
        """
        Args:
            texts: [batch_size, seq_len] token ids
        Returns:
            text_features: [batch_size, embed_dim] 归一化后
        """
        # 通过文本编码器（简化，实际需用 Transformer）
        text_features = self.text_encoder(texts)
        # 取 [CLS] token 或 pooling
        text_features = text_features[:, 0, :]  # 假设第一维是 CLS
        # 投影
        text_features = text_features @ self.text_projection
        # L2 归一化
        text_features = nn.functional.normalize(text_features, p=2, dim=-1)
        return text_features
    
    def forward(self, images: torch.Tensor, texts: torch.Tensor):
        """
        计算对比学习损失
        """
        image_features = self.encode_image(images)
        text_features = self.encode_text(texts)
        
        # 计算相似度矩阵 [batch, batch]
        logit_scale = self.logit_scale.exp()
        logits_per_image = logit_scale * image_features @ text_features.t()
        logits_per_text = logits_per_image.t()
        
        # 对比学习损失（对称交叉熵）
        batch_size = logits_per_image.size(0)
        labels = torch.arange(batch_size, device=images.device)
        
        loss_image = nn.functional.cross_entropy(logits_per_image, labels)
        loss_text = nn.functional.cross_entropy(logits_per_text, labels)
        loss = (loss_image + loss_text) / 2
        
        return loss, logits_per_image, logits_per_text
```

#### 题目 5：实现 LoRA 高效微调

```python
class LoRALinear(nn.Module):
    """
    LoRA 线性层
    考点：理解低秩分解、参数高效微调
    """
    def __init__(self, in_features: int, out_features: int, 
                 rank: int = 8, alpha: float = 16.0, 
                 dropout: float = 0.1, bias: bool = False):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.rank = rank
        self.alpha = alpha
        self.scaling = alpha / rank
        
        # 原始权重（冻结）
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        if bias:
            self.bias = nn.Parameter(torch.empty(out_features))
        else:
            self.register_parameter('bias', None)
        
        # LoRA 适配器（可训练）
        # A: 高斯初始化，B: 零初始化（保证初始为恒等映射）
        self.lora_A = nn.Parameter(torch.empty(rank, in_features))
        self.lora_B = nn.Parameter(torch.empty(out_features, rank))
        
        self.dropout = nn.Dropout(dropout)
        self._init_weights()
    
    def _init_weights(self):
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        nn.init.kaiming_uniform_(self.lora_A, a=math.sqrt(5))
        nn.init.zeros_(self.lora_B)
        if self.bias is not None:
            nn.init.zeros_(self.bias)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # 原始前向传播
        result = nn.functional.linear(x, self.weight, self.bias)
        
        # LoRA 分支：x @ A^T @ B^T * scaling
        # x: [batch, seq, in_features]
        # lora_A: [rank, in_features] -> 需要转置
        # lora_B: [out_features, rank]
        lora_out = self.dropout(x) @ self.lora_A.t() @ self.lora_B.t()
        result = result + lora_out * self.scaling
        
        return result
    
    def merge_weights(self) -> nn.Linear:
        """
        合并 LoRA 权重到原始权重（推理时使用）
        """
        merged_weight = self.weight + (self.lora_B @ self.lora_A) * self.scaling
        merged_linear = nn.Linear(self.in_features, self.out_features, 
                                  bias=self.bias is not None)
        merged_linear.weight = nn.Parameter(merged_weight)
        if self.bias is not None:
            merged_linear.bias = self.bias
        return merged_linear


class LoRAForTransformer(nn.Module):
    """
    在 Transformer 中应用 LoRA
    考点：知道应该对哪些层应用 LoRA
    """
    def __init__(self, base_model: nn.Module, rank: int = 8, 
                 target_modules: list = None):
        """
        Args:
            base_model: 预训练的 Transformer 模型
            rank: LoRA 秩
            target_modules: 要应用 LoRA 的模块名，如 ['q_proj', 'v_proj']
        """
        super().__init__()
        self.base_model = base_model
        self.rank = rank
        
        if target_modules is None:
            target_modules = ['q_proj', 'v_proj']  # 默认对 Q、V 投影应用
        
        self._apply_lora(base_model, target_modules, rank)
    
    def _apply_lora(self, module: nn.Module, target_modules: list, rank: int):
        for name, child in module.named_children():
            if name in target_modules and isinstance(child, nn.Linear):
                # 替换为 LoRA 线性层
                lora_layer = LoRALinear(
                    child.in_features, 
                    child.out_features,
                    rank=rank
                )
                # 复制原始权重
                lora_layer.weight.data = child.weight.data.clone()
                if child.bias is not None:
                    lora_layer.bias.data = child.bias.data.clone()
                setattr(module, name, lora_layer)
            else:
                # 递归处理子模块
                self._apply_lora(child, target_modules, rank)
    
    def forward(self, *args, **kwargs):
        return self.base_model(*args, **kwargs)
    
    def get_lora_parameters(self):
        """只返回 LoRA 参数用于优化"""
        lora_params = []
        for name, param in self.named_parameters():
            if 'lora_' in name:
                lora_params.append(param)
```

```python
import torchvision.transforms as T
from PIL import Image
import numpy as np

class VideoFrameSampler:
    """
    视频帧采样策略
    考点：理解不同采样策略的适用场景
    """
    def __init__(self, num_frames: int = 16, sampling_strategy: str = 'uniform'):
        self.num_frames = num_frames
        self.sampling_strategy = sampling_strategy
    
    def sample(self, total_frames: int) -> np.ndarray:
        """
        Args:
            total_frames: 视频总帧数
        Returns:
            frame_indices: 采样的帧索引
        """
        if self.sampling_strategy == 'uniform':
            # 均匀采样
            indices = np.linspace(0, total_frames - 1, self.num_frames, dtype=int)
        elif self.sampling_strategy == 'random':
            # 随机采样
            indices = np.random.choice(total_frames, self.num_frames, replace=False)
            indices = np.sort(indices)
        elif self.sampling_strategy == 'dense':
            # 密集采样（短片段）
            if total_frames <= self.num_frames:
                indices = np.arange(total_frames)
            else:
                start = np.random.randint(0, total_frames - self.num_frames)
                indices = np.arange(start, start + self.num_frames)
        else:
            raise ValueError(f"Unknown strategy: {self.sampling_strategy}")
        
        return indices


class VideoTransform:
    """
    视频预处理变换
    考点：理解视频与图像预处理的区别
    """
    def __init__(self, img_size: int = 224, mean: list = None, std: list = None):
        self.img_size = img_size
        self.mean = mean or [0.485, 0.456, 0.406]
        self.std = std or [0.229, 0.224, 0.225]
        
        self.transform = T.Compose([
            T.Resize(img_size),
            T.CenterCrop(img_size),
            T.ToTensor(),
            T.Normalize(mean=self.mean, std=self.std)
        ])
    
    def __call__(self, frames: list) -> torch.Tensor:
        """
        Args:
            frames: list of PIL.Image 或 np.ndarray
        Returns:
            video_tensor: [T, C, H, W]
        """
        transformed = [self.transform(frame) for frame in frames]
        return torch.stack(transformed, dim=0)
```

#### 题目 7：实现时序注意力（TimeSformer 简化版）

```python
class TemporalAttention(nn.Module):
    """
    时序注意力机制
    考点：理解时空分离的注意力计算
    """
    def __init__(self, d_model: int, num_frames: int, num_heads: int = 8):
        super().__init__()
        self.num_frames = num_frames
        self.temporal_attn = nn.MultiheadAttention(d_model, num_heads, batch_first=True)
        self.norm = nn.LayerNorm(d_model)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: [batch, num_frames*num_patches, d_model]
               或 [batch, num_frames, num_patches, d_model]
        Returns:
            x: 时序注意力处理后的特征
        """
        # 假设输入是 [batch, num_frames, num_patches, d_model]
        B, T, N, D = x.shape
        
        # 对每个 patch，在时序维度上做注意力
        # 重排为 [batch*num_patches, num_frames, d_model]
        x = x.permute(0, 2, 1, 3).contiguous().view(B * N, T, D)
        
        # 时序自注意力
        attn_out, _ = self.temporal_attn(x, x, x)
        
        # 残差 + 归一化
        x = self.norm(x + attn_out)
        
        # 恢复形状 [batch, num_frames, num_patches, d_model]
        x = x.view(B, N, T, D).permute(0, 2, 1, 3)
        
        return x


class DividedSpaceTimeAttention(nn.Module):
    """
    分离的时空注意力（TimeSformer 核心）
    考点：理解为什么分离时空注意力更高效
    """
    def __init__(self, d_model: int, num_heads: int, num_frames: int):
        super().__init__()
        self.temporal_attn = nn.MultiheadAttention(d_model, num_heads, batch_first=True)
        self.spatial_attn = nn.MultiheadAttention(d_model, num_heads, batch_first=True)
        self.norm1 = nn.LayerNorm(d_model)
        self.norm2 = nn.LayerNorm(d_model)
        self.num_frames = num_frames
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x: [batch, num_frames*num_patches+1, d_model]  # +1 for CLS
        Returns:
            x: 时空注意力处理后的特征
        """
        B, L, D = x.shape
        T = self.num_frames
        N = L - 1  # 去掉 CLS token
        patches_per_frame = N // T
        
        # 分离 CLS 和 patch tokens
        cls_token = x[:, 0:1, :]  # [B, 1, D]
        patch_tokens = x[:, 1:, :]  # [B, N, D]
        
        # 重排为 [B, T, N/T, D]
        patch_tokens = patch_tokens.view(B, T, patches_per_frame, D)
        
        # 1. 时序注意力（对每个 patch 在时间维度）
        x_temporal = patch_tokens.permute(0, 2, 1, 3).contiguous()  # [B, N/T, T, D]
        x_temporal = x_temporal.view(B * patches_per_frame, T, D)
        attn_out, _ = self.temporal_attn(x_temporal, x_temporal, x_temporal)
        x_temporal = self.norm1(x_temporal + attn_out)
        x_temporal = x_temporal.view(B, patches_per_frame, T, D)
        
        # 2. 空间注意力（对每帧在空间维度）
        x_spatial = x_temporal.permute(0, 2, 1, 3).contiguous()  # [B, T, N/T, D]
        x_spatial = x_spatial.view(B * T, patches_per_frame, D)
        attn_out, _ = self.spatial_attn(x_spatial, x_spatial, x_spatial)
        x_spatial = self.norm2(x_spatial + attn_out)
        
        # 恢复形状
        x_spatial = x_spatial.view(B, T, patches_per_frame, D)
        x_spatial = x_spatial.view(B, T * patches_per_frame, D)
        
        # 拼接 CLS
        x = torch.cat([cls_token, x_spatial], dim=1)
        
        return x
```

### 2.4 分布式训练实现

#### 题目 8：实现 DDP 训练循环

```python
import torch.distributed as dist
from torch.nn.parallel import DistributedDataParallel as DDP
from torch.utils.data.distributed import DistributedSampler

def setup_distributed(rank: int, world_size: int):
    """初始化分布式环境"""
    dist.init_process_group("nccl", rank=rank, world_size=world_size)
    torch.cuda.set_device(rank)

def cleanup_distributed():
    """清理分布式环境"""
    dist.destroy_process_group()

class DistributedTrainer:
    """
    DDP 训练器
    考点：理解 DDP 工作原理、梯度同步、采样器
    """
    def __init__(self, model: nn.Module, train_dataset, 
                 rank: int, world_size: int, batch_size: int = 32):
        self.rank = rank
        self.world_size = world_size
        
        # 分布式采样器（保证每个 GPU 看到不同的数据）
        sampler = DistributedSampler(train_dataset, num_replicas=world_size, 
                                     rank=rank, shuffle=True)
        
        self.dataloader = torch.utils.data.DataLoader(
            train_dataset,
            batch_size=batch_size,
            sampler=sampler,
            num_workers=4,
            pin_memory=True
        )
        
        # 模型移到 GPU 并包装 DDP
        model = model.cuda(rank)
        self.model = DDP(model, device_ids=[rank], 
                        find_unused_parameters=False)
        
        self.optimizer = torch.optim.AdamW(self.model.parameters(), lr=1e-4)
        self.scaler = torch.cuda.amp.GradScaler()  # 混合精度
    
    def train_epoch(self, epoch: int):
        self.model.train()
        self.dataloader.sampler.set_epoch(epoch)  # 重要！保证 shuffle
        
        for batch_idx, (data, target) in enumerate(self.dataloader):
            data, target = data.cuda(self.rank), target.cuda(self.rank)
            
            # 混合精度训练
            with torch.cuda.amp.autocast():
                output = self.model(data)
                loss = nn.functional.cross_entropy(output, target)
            
            self.optimizer.zero_grad()
            self.scaler.scale(loss).backward()
            self.scaler.step(self.optimizer)
            self.scaler.update()
            
            # 每 100 batch 打印（只在 rank 0 打印）
            if batch_idx % 100 == 0 and self.rank == 0:
                print(f"Epoch {epoch}, Batch {batch_idx}, Loss: {loss.item():.4f}")
    
    def save_checkpoint(self, path: str, epoch: int):
        """只在 rank 0 保存"""
        if self.rank == 0:
            checkpoint = {
                'epoch': epoch,
                'model_state_dict': self.model.module.state_dict(),
                'optimizer_state_dict': self.optimizer.state_dict(),
            }
            torch.save(checkpoint, path)
            print(f"Checkpoint saved to {path}")


# 启动脚本示例
def spawn_ddp_training():
    """使用 torch.multiprocessing.spawn 启动多进程"""
    import torch.multiprocessing as mp
    
    def worker(rank, world_size):
        setup_distributed(rank, world_size)
        
        # 创建模型和数据集
        model = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(d_model=512, nhead=8),
            num_layers=6
        )
        dataset = torch.utils.data.TensorDataset(
            torch.randn(10000, 3, 224, 224),
            torch.randint(0, 10, (10000,))
        )
        
        trainer = DistributedTrainer(model, dataset, rank, world_size, batch_size=32)
        
        for epoch in range(10):
            trainer.train_epoch(epoch)
            trainer.save_checkpoint(f'checkpoint_epoch{epoch}.pt', epoch)
        
        cleanup_distributed()
    
    world_size = torch.cuda.device_count()
    mp.spawn(worker, args=(world_size,), nprocs=world_size, join=True)
```

### 3.1 多模态对齐技术详解

#### 对比学习对齐（CLIP 范式）

**核心思想：**
- 将图像和文本映射到同一嵌入空间
- 匹配的图文对相似度最大化，不匹配的 minimiz

**损失函数：**
```python
def clip_loss(image_emb, text_emb, temperature=0.07):
    # image_emb: [B, D], text_emb: [B, D]
    # 归一化
    image_emb = F.normalize(image_emb, p=2, dim=1)
    text_emb = F.normalize(text_emb, p=2, dim=1)
    
    # 相似度矩阵
    logit_scale = np.log(1 / temperature)
    logits = image_emb @ text_emb.t() * logit_scale
    
    # 对称交叉熵
    labels = torch.arange(len(logits))
    loss_i2t = F.cross_entropy(logits, labels)
    loss_t2i = F.cross_entropy(logits.t(), labels)
    
    return (loss_i2t + loss_t2i) / 2
```

**关键技巧：**
1. 温度参数可学习
2. 大 batch size（4096+）对对比学习至关重要
3. 数据增强对图像端很重要（RandomCrop、ColorJitter）

#### 生成式对齐（BLIP 范式）

**核心思想：**
- 使用生成任务（如 Image Captioning）进行对齐
- 多任务学习：ITM（图文匹配）+ ITG（图文生成）+ ITC（对比学习）

**优势：**
- 比纯对比学习更灵活
- 支持开放域生成任务
- 可以处理更复杂的跨模态推理

### 3.2 视频理解关键技术

#### 时序建模方法对比

| 方法 | 优点 | 缺点 | 适用场景 |
|------|------|------|---------|
| 3D CNN | 局部时空特征好 | 长程依赖弱 | 短视频动作识别 |
| LSTM/GRU | 序列建模成熟 | 并行化差 | 早期视频理解 |
| Transformer | 长程依赖强 | 计算复杂度高 | 长视频理解 |
| TimeSformer | 时空分离高效 | 需要调参 | 通用视频任务 |

#### 视频预训练任务

1. **帧排序预测**：打乱帧顺序，预测正确顺序
2. **掩码帧预测**：类似 BERT，掩码部分帧预测
3. **对比学习**：同视频片段为正样本，不同为负
4. **多模态对齐**：视频 - 文本对比学习

### 3.3 大模型高效微调技术对比

| 方法 | 可训练参数 | 性能 | 显存占用 | 适用场景 |
|------|-----------|------|---------|---------|
| Full FT | 100% | 最佳 | 最高 | 小模型、充足资源 |
| LoRA | 0.1-1% | 接近 Full | 低 | 大模型微调 |
| Adapter | 1-5% | 良好 | 中 | 多任务学习 |
| Prefix Tuning | 0.01-0.1% | 较好 | 最低 | 生成任务 |

### 3.4 分布式训练策略

#### 数据并行（DDP）
- 每个 GPU 持有一份完整模型
- 数据分片，梯度同步
- 适合中小模型

#### 模型并行（Megatron-LM）
- 模型切分到多个 GPU
- 适合超大模型（>10B）

#### ZeRO 优化（DeepSpeed）
- ZeRO-1：优化器状态分片
- ZeRO-2：优化器 + 梯度分片
- ZeRO-3：优化器 + 梯度 + 参数分片
